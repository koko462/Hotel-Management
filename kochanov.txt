using System;
using System.Diagnostics;
using System.Threading.Tasks;

class Program
{
    static async Task Main(string[] args)
    {
        await Demo();
    }

    static async Task Demo()
    {
        var stopwatch = Stopwatch.StartNew();

        Console.WriteLine("Work Started for 5 sec.");
        await StartingSomeWork();

        var worker1Task = Worker1();
        var worker2Task = Worker2();

        await Task.WhenAll(worker1Task, worker2Task);

        Console.WriteLine("Work Ended");

        stopwatch.Stop();
        Console.WriteLine($"Execution Time: {stopwatch.ElapsedMilliseconds} ms");

        Console.WriteLine("Press any key to continue...");
        Console.ReadKey();
    }

    static async Task StartingSomeWork()
    {
        await Task.Delay(5000);
    }

    static async Task Worker1()
    {
        Console.WriteLine("Worker1 does some work for 2 sec.");
        await Task.Delay(2000);
        Console.WriteLine("Worker1 end some work");
    }

    static async Task Worker2()
    {
        Console.WriteLine("Worker2 does some work for 3 sec.");
        await Task.Delay(3000);
        Console.WriteLine("Worker2 end some work");
    }
}

using System;
using System.Diagnostics;
using System.Threading.Tasks;

class Program
{
    static async Task Main(string[] args)
    {
        Console.WriteLine("Welcome to the ticket booking system");
        Task bookingTask1 = BookAsync(1, "Ivanov");
        Task bookingTask2 = BookAsync(2, "Petrov");
        Task.WaitAll(bookingTask1, bookingTask2);
        Console.WriteLine("All tickets booked succesfully!");
    }
    static async Task BookAsync(int ticketId, string customerName)
    {
        Console.WriteLine($"Booking ticked {ticketId} for {customerName}");
        await Task.Delay(1000);
        Console.WriteLine($"Ticket {ticketId} booked for {customerName}");
    }

}
